\begin{algorithm}
    \KwIn{content}
    \KwOut{fixed\_content}

    words $\gets$ Split content into list of words using whitespace as delimiter\;
    replaced\_words $\gets$ [ ]\;

    \ForEach{word $\in$ words}{
        \eIf{word exist as a key inside WORD\_FIX\_DICT}{
            replaced\_word $\gets$ WORD\_FIX\_DICT[word]\;
            replaced\_words.append(replaced\_word)\;
        }{
            replaced\_words.append(word)\;
        }
    }
    fixed\_content $\gets$ Join replaced\_words into a single string with spaces\;
    \Return fixed\_content\;
    \caption{FixWordsByDict: Fix word-level noise}
    \label{alg:word_level_noise_fix}
\end{algorithm}


\begin{algorithm}
    \SetKwInOut{Input}{Input}
    \SetKwInOut{Output}{Output}

    \Input{content}
    \Output{fixed\_content}

    fixed\_content $\gets$ content\;

    \ForEach{$(\text{word}, \text{fixed\_word})$ \textbf{in} PHRASE\_NOISE\_DICT}{
        fixed\_content $\gets$ fixed\_content.replace(word, fixed\_word)\;
    }

    \Return{fixed\_content}
    \caption{RemoveNoisePhrases: Remove phrase-level noise}
    \label{alg:phrase_level_noise_fix}
\end{algorithm}
